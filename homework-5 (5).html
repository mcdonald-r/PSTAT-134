import numpy as np

def give_recommendation(movie_title, movies, similarity_matrix):
    movie_index = movies.index[movies['title'] == movie_title][0]

    similarity_scores = similarity_matrix[movie_index]

    top_indices = np.argsort(similarity_scores)[::-1][1:11]

    recommendations = [(movies.iloc[i]['title'], similarity_scores[i]) for i in top_indices]

    return recommendations
